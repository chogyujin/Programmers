class Solution {
    public int[] solution(int n) {
        int temp = 0;
        int tempn = n;
        for(int i=0;i<n;i++){
            temp+=tempn;
            tempn--;
        }
        int[] answer = new int[temp];//answer배열 크기 설정
        
        //삼각형배열그리기(정확힌 직각삼각형)
        int[][] arr = new int[n+1][n+1];
        int[] dx = {0,1,-1};
        int[] dy = {1,0,-1};
        tempn = n;
        int x = 0;
        int y = 0;
        int dir = -1;
        int cnt = 1;
        while(n>0){
            dir = (dir+1)%3;
            for(int i=0;i<n;i++){
                x = x+dx[dir];
                y = y+dy[dir];
                arr[y][x] = cnt;
                cnt++;
            }
            n--;
        }
        
        //answer배열에 셋팅
        cnt = 0;
        for(int i=0;i<tempn+1;i++){
            for(int j=0;j<tempn+1;j++){
                if(arr[i][j]!=0){
                    answer[cnt] = arr[i][j];   
                    cnt++;
                }
            }
        }
        return answer;
    }
}
