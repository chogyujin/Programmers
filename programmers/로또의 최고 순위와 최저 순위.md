class Solution {
    public int[] solution(int[] lottos, int[] win_nums) {
        int zeroCnt = 0;
        int cnt = 0;
        
        for(int i : lottos){
            if(i==0) zeroCnt++;
            for(int j : win_nums){
                if(i==j){
                    cnt++;
                    break;
                }
            }
        }
        
        int bestRnk = rankDiscriminate(cnt+zeroCnt);
        int worstRnk = rankDiscriminate(cnt);
        
        int[] answer = {bestRnk,worstRnk};
        
        return answer;
    }

    private int rankDiscriminate(int cnt){
        int rnk=0;
        if(cnt >= 6) cnt = 6;
        switch(cnt){
            case 6 : rnk = 1;
                break;
            case 5 : rnk = 2;
                break;
            case 4 : rnk = 3;
                break;
            case 3 : rnk = 4;
                break;
            case 2 : rnk = 5;
                break;
            default : rnk = 6;
        }
        
        return rnk;
    }
}
