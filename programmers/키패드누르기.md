class Solution {
    
    StringBuilder sb = new StringBuilder();
    int[] lh = {3,0};
    int[] rh = {3,2};
    int[][] keypad = {{3,1},//0
                    {0,0},{0,1},{0,2},//1,2,3
                    {1,0},{1,1},{1,2},//4,5,6
                    {2,0},{2,1},{2,2}};//7,8,9
    
    public String solution(int[] numbers, String hand) {
        String answer = "";
        
        for(int i=0;i<numbers.length;i++){
            if(numbers[i]==1 || numbers[i]==4 || numbers[i]==7){
                keyPush("L",numbers,i);
            }
            else if(numbers[i]==3 || numbers[i]==6 || numbers[i]==9){
                keyPush("R",numbers,i);
            }
            else{
                int leftLen = Math.abs(lh[0]-keypad[numbers[i]][0])+Math.abs(lh[1]-keypad[numbers[i]][1]);
                int rightLen = Math.abs(rh[0]-keypad[numbers[i]][0])+Math.abs(rh[1]-keypad[numbers[i]][1]);
                if(leftLen<rightLen){
                    keyPush("L",numbers,i);
                }
                else if(leftLen>rightLen){
                    keyPush("R",numbers,i);
                }
                else{
                    if(hand.equals("left")){
                        keyPush("L",numbers,i);
                    }
                    else{
                        keyPush("R",numbers,i);
                    }
                }
            }
        }
        answer = sb.toString();
        return answer;
    }
    
    
    private int keyPush(String handD,int[] numbers,int i){
        if(handD=="L"){
            sb.append("L");
            lh[0] = keypad[numbers[i]][0];
            lh[1] = keypad[numbers[i]][1];
        }
        else{
            sb.append("R");
            rh[0] = keypad[numbers[i]][0];
            rh[1] = keypad[numbers[i]][1];
        }
        return 0;
    }
}
