```SQL
SELECT  A.*
FROM    PLACES A JOIN (SELECT HOST_ID
        FROM    PLACES
        GROUP BY HOST_ID
        HAVING COUNT(*) >= 2) B ON A.HOST_ID = B.HOST_ID
ORDER BY A.ID
```

```SQL
SELECT  DISTINCT(A.ID),A.NAME,A.HOST_ID
FROM    PLACES A JOIN 
        (SELECT HOST_ID
        FROM    PLACES
        GROUP BY HOST_ID HAVING COUNT(*)>=2) B
        ON A.HOST_ID = B.HOST_ID
ORDER BY A.ID,A.NAME,A.HOST_ID
```
